Detection Engineering

Cross-platform rules mapped to MITRE ATT&CK. Source files in detections/.

<rule id="180001" level="8">
  <match>bitsadmin.exe</match>
  <match type="pcre2">(?i)/transfer.*/download</match>
  <mitre><id>T1197</id></mitre>
</rule>
index=sysmon EventCode=1
| search Image="*\\bitsadmin.exe" CommandLine="*transfer*"
| eval mitre_technique="T1197"
| table _time host User CommandLine ParentImage
DeviceProcessEvents
| where FileName =~ "bitsadmin.exe"
| where ProcessCommandLine has_all ("transfer", "download")
| extend MITRE_Technique = "T1197"
| project Timestamp, DeviceName, ProcessCommandLine
